NoticeZ.js


A cool way to make people notice

Examples:

Click on the button below to see the default notice box !

$("#btn1").click(function(){
	var options = {
		title : "Hey !",
		body  : "This is a default notice box!<br>It will fade out in 3 sec"
	}
	noticeZ(options);
});

Successful:

Click on the button below to see the success notice box !

$("#btn2").click(function(){
	var options = {
		title : "Hey !",
		body  : "This notice box indicates a successful or positive action and it's on the left",
		state : "success",
		position: "topLeft"
	}
	noticeZ(options);
});

Warning:

Click on the button below to see the warning notice box !

$("#btn3").click(function(){
	var options = {
		title : "Oops !",
		body  : "Something wrong",
		state : "warning",
		position: "bottomLeft",
		font  : "Raleway"
	}
	noticeZ(options);
});

Info:

Click on the button below to see the info notice box !

$("#btn4").click(function(){
	var options = {
		title : "Hey !",
		body  : "This notice box show some informations and it will fade out in 2sec",
		state : "info",
		position: "bottomRight",
		font  : "Raleway",
		time  : 2000 
	}
	noticeZ(options);
});

Animate:

If you're using animate.css library by daneden, you can even do this !

$("#btn5").click(function(){
	var options = {
		title : "Hey !",
		body  : "This box will dissappear with difference animation",
		state : "success",
		position: "topRight",
		font  : "Raleway",
		time  : 2000,
		animate: "slideOutRight" 
	}
	noticeZ(options);
});
